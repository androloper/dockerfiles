docker login (login from docker desktop or docherhub to make it easier)
docker logout

docker build -t nickname/repositoryname:versionname .  (dont forget to .(dot) if you arent using versionname it comes latest as a default)

docker push nickname/repositoryname:versionname (for pushing to the cloud(docker hub))

docker pull nickname/repositoryname:versionname (for pulling from the cloud(docker hub))

(if http is active)
docker run --rm -it [--restart always] [--name containername] [-p (port out):(port in)] [--env connectionstring='database path'] [--env DOTNET_SYSTEM_GLOBALIZATION_INVARIANT=false] -d nickname/repositoryname:versionname (If you open docker desktop in startup, it automatically makes it active when the computer opens.)

(if https is active)
docker run —rm -it [—name containername] [-p (portout):(portin)][-e ASPNETCORE_URLS=“https://+;http://+”] [-e ASPNETCORE_HTTPS_PORT=(portout)] [-e ASPNETCORE_Kestrel__Certificates__Default__Password=“pass”] [-e ASPNETCORE_Kestrel__Certificates__Default__Path=/https/ssl.pfx] [-v path] username/repositoryname:versionname
